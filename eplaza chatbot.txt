    def _create_system_prompt(self) -> str:
        product_catalog = "COMPLETE PRODUCT CATALOG:\n\n"
        
        for category, products in self.products.items():
            category_name = category.replace('_', ' ').title()
            product_catalog += f"=== {category_name.upper()} ===\n"
            
            for product in products:
                product_catalog += f"â€¢ {product['name']}\n"
                product_catalog += f"  Price: Rs. {product['price']:,}\n"
                product_catalog += f"  Brand: {product['brand']}\n"
                product_catalog += f"  Description: {product['description']}\n"
                product_catalog += f"  Features: {', '.join(product['features'])}\n\n"
        
        return f"""You are an expert product consultant and sales assistant for an e-commerce store. You have COMPLETE access to our entire product database below.

{product_catalog}

YOUR ROLE:
- Product expert who knows every item in our catalog
- Friendly shopping assistant who helps customers find perfect products
- Comparison specialist who can analyze differences between products
- Budget advisor who suggests best value options
- Style consultant who understands customer needs

CONVERSATION STYLE:
- Be conversational, friendly, and helpful like a knowledgeable friend
- Use PLAIN TEXT only - NO asterisks, bold, or special formatting
- Keep responses concise but informative (3-5 sentences max)
- Always mention specific product names and exact prices in PKR
- Compare products when asked, highlighting key differences
- Suggest alternatives based on budget or preferences
- Ask follow-up questions to better understand customer needs

CORE CAPABILITIES:
1. PRODUCT RECOMMENDATIONS: Suggest best products based on user needs/budget
2. COMPARISONS: Compare any products side-by-side with pros/cons
3. BUDGET ANALYSIS: Find best value within price ranges
4. FEATURE EXPLANATION: Explain what makes products special
5. STYLE ADVICE: Help match products to customer lifestyle

RESPONSE FORMAT:
- Start with direct answer to user question
- Mention 1-3 specific products with exact prices
- Give brief reason why each product is good
- End with helpful follow-up question

Remember: You know ALL our products intimately. Be confident, helpful, and conversational!"""

    def process_query(self, user_input: str, session_id: str = "default") -> str:
        if session_id not in self.conversations:
            self.conversations[session_id] = []
        
        greetings = ['hi', 'hello', 'hey', 'sup', 'hii', 'helo']
        if user_input.lower().strip() in greetings:
            response = "Hi there! I'm your product expert. I know all about our earbuds, smartphones, clothing, fragrances, skincare, smartwatches, and jeans. What can I help you find today?"
            self.conversations[session_id].append({'user': user_input, 'assistant': response})
            return response
        
        if len(user_input.strip()) < 2:
            response = "I'm here to help you find the perfect products! Ask me about anything - comparisons, recommendations, or specific items you're looking for."
            self.conversations[session_id].append({'user': user_input, 'assistant': response})
            return response
        
        messages = [
            {"role": "system", "content": self.system_prompt}
        ]
        
        recent_history = self.conversations[session_id][-6:] if len(self.conversations[session_id]) > 6 else self.conversations[session_id]
        for msg in recent_history:
            messages.append({"role": "user", "content": msg['user']})
            messages.append({"role": "assistant", "content": msg['assistant']})
        
        messages.append({"role": "user", "content": user_input})
        
        try:
            headers = {
                "Authorization": f"Bearer {self.api_key}",
                "Content-Type": "application/json"
            }
            
            payload = {
                "model": self.model,
                "messages": messages,
                "temperature": 0.4,
                "max_tokens": 400,
                "top_p": 0.8
            }
            
            response = requests.post(self.base_url, json=payload, headers=headers, timeout=30)
            response.raise_for_status()
            
            result = response.json()
            
            if 'choices' not in result or not result['choices']:
                response_text = "Sorry, I didn't get a proper response. What specific product are you interested in?"
            else:
                response_text = result['choices'][0]['message']['content'].strip()
                
                if not response_text or len(response_text) < 10:
                    response_text = "I'm your product expert! Ask me about earbuds, phones, clothes, perfumes, skincare, watches, or jeans. What interests you?"
            
            self.conversations[session_id].append({'user': user_input, 'assistant': response_text})
            
            if len(self.conversations[session_id]) > 12:
                self.conversations[session_id] = self.conversations[session_id][-10:]
            
            return response_text
            
        except requests.exceptions.Timeout:
            response_text = "Response took too long. Ask me about a specific product and I'll help quickly!"
        except requests.exceptions.RequestException:
            response_text = "Having trouble connecting. Ask me about any product - earbuds, phones, clothing, or others!"
        except Exception:
            response_text = "Something went wrong. What product are you interested in? I can help with recommendations and comparisons!"
        
        self.conversations[session_id].append({'user': user_input, 'assistant': response_text})
        return response_text

@app.route('/api/chat', methods=['POST'])
def chat():
    try:
        data = request.get_json()
        user_message = data.get('message', '').strip()
        session_id = data.get('session_id', 'web_session')
        
        if not user_message:
            return jsonify({'error': 'No message provided'}), 400
        
        response = chatbot.process_query(user_message, session_id)
        
        return jsonify({
            'response': response,
            'session_id': session_id,
            'timestamp': datetime.datetime.now().isoformat()
        })
        
    except Exception as e:
        return jsonify({'error': f'An error occurred: {str(e)}'}), 500
